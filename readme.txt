args:
    --inFrameStart int64
        first frame number

    --inFrameEnd int64
        last frame number

    --threadCount int
        thread count

    --invert
        inverts alpha

    --quadMinSize int
        min width/height of quadrant

    --quadTolerance int
        color diff tolerance

    --repRepeat int
        rep frame repeat count

    --inDir string
        input dir

    --outDir string
        output dir

    --repDir string
        replace frame dir
